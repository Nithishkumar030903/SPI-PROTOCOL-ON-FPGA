module clock_divider(
    input clk,
    input reset,
    input [15:0] divider,
    output reg sclk
);

reg [15:0] count;

always @(posedge clk or posedge reset) begin
    if (reset) begin
        count <= 0;
        sclk <= 0;
    end 
    else begin
        if (count == divider) begin
            count <= 0;
            sclk <= ~sclk;
        end 
        else begin
            count <= count + 1;
        end
    end
end

endmodule
